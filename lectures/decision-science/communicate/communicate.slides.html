    """This is the function docstring"""
    print('required: ', required)
    if args:

    if kwargs:

Python decorators¶

In [138]:
def my_decorator(func):
    def wrapper(something):
        print('===================')
        print(f"| {func(something)} |")
        print('===================')

    return wrapper

@my_decorator # will do say = my_decorator(say) after the definition
def say(something):
    return something


# say = my_decorator(say)

say('hello everyone!')
